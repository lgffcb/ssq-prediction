# 双色球数据自动更新 - crontab 配置

# 每天早上 9 点检查更新（双色球开奖日是周二、四、日）
# 在终端执行：crontab -e，然后添加以下内容

# 环境变量
SHELL=/bin/bash
PATH=/usr/local/bin:/usr/bin:/bin
PYTHON=/usr/bin/python3

# 每天 9:00 检查并更新
0 9 * * * cd /home/admin/openclaw/workspace/lottery-predictor && $PYTHON auto_update.py -f data/historical_data.csv -s mock -i daily >> logs/auto_update.log 2>&1

# 或者每周检查一次（每周一 9:00）
# 0 9 * * 1 cd /home/admin/openclaw/workspace/lottery-predictor && $PYTHON auto_update.py -f data/historical_data.csv -s mock -i weekly >> logs/auto_update.log 2>&1

# 开奖日晚上 9 点检查（周二、四、日）
# 0 21 * * 2,4,0 cd /home/admin/openclaw/workspace/lottery-predictor && $PYTHON auto_update.py -f data/historical_data.csv -s mock -i always >> logs/auto_update.log 2>&1

# 静默模式（只在有更新时输出）
# 0 9 * * * cd /home/admin/openclaw/workspace/lottery-predictor && $PYTHON auto_update.py -f data/historical_data.csv -s mock -i daily -q >> logs/auto_update.log 2>&1

# 使用示例：
# 1. 创建日志目录
#    mkdir -p /home/admin/openclaw/workspace/lottery-predictor/logs
#
# 2. 编辑 crontab
#    crontab -e
#
# 3. 添加上面的配置（选择一行即可）
#
# 4. 查看 crontab
#    crontab -l
#
# 5. 查看日志
#    tail -f /home/admin/openclaw/workspace/lottery-predictor/logs/auto_update.log
